Basic inset example


import matplotlib.pyplot as plt
import numpy as np

x = np.linspace(0, 10, 500)
y = np.sin(x) * np.exp(-0.1 * x)

fig, ax = plt.subplots()
ax.plot(x, y, color="steelblue")
ax.set_title("Inset and Secondary Views")

# Create an inset axis: [left, bottom, width, height] in relative (0â€“1) coordinates
ins = ax.inset_axes([0.6, 0.55, 0.35, 0.35])
ins.plot(x, y, color="darkorange")
ins.set_title("Inset", fontsize=9)
plt.show()

The coordinates [0.6, 0.55, 0.35, 0.35] place the inset inside the parent axes, expressed as a fraction of the main plotâ€™s width and height.

Zooming into a region


# Highlight a region in the inset
x_zoom = x[(x > 2) & (x < 4)]
y_zoom = y[(x > 2) & (x < 4)]
ins.plot(x_zoom, y_zoom, color="darkorange")
ins.set_xlim(2, 4)
ins.set_ylim(-0.5, 1.0)

You can visually connect the inset to the zoomed area using ax.indicate_inset_zoom(). 


from mpl_toolkits.axes_grid1.inset_locator import inset_axes, mark_inset

# Using inset_locator utilities for connection lines
ins = inset_axes(ax, width="35%", height="35%", loc="upper right")
ins.plot(x, y, color="darkorange")
ax.indicate_inset_zoom(ins, edgecolor="gray")

Secondary axes (shared data)


# Secondary axis for alternative units
def mph_to_kph(x): return x * 1.609
def kph_to_mph(x): return x / 1.609

secax = ax.secondary_xaxis('top', functions=(kph_to_mph, mph_to_kph))
secax.set_xlabel("Speed (mph)")
